Pointer Arithmetic

Comparando ponteiros para comparar ponteiros podemos usar o exemplo abaixo

#include <iostream>
#include <string>


using namespace std;


int main(){


    string s1 {"Frank"};
    string s2 {"Frank"};


    string *p1 {&s1};
    string *p2 {&s2};
    string *p3 {&s1};


    cout << (p1 == p2) << endl;
    cout << (p1 == p3) << endl;
    return 0;



}

Looping um array usando ponteiros

Quando usamos o operador de incremento nos ponteiros ++
ele não incrementa o valor do ponteiro e sim o valor do tipo do dado numero exemplo
no meu compudador um **int** tem 4 bites então ele vai incrementar a o ponteiro em 4 bites
então como os itens do array são guardados em seguncia ele vai fazer o loop na memoria aumentando o endereço de 4 em 4 
#include <iostream>
#include <string>


using namespace std;


int main(){


    int score[]{100,95,89,68,-1};
    int *score_ptr{score};


    while (*score_ptr != -1)
    {
        cout << *score_ptr << endl;
        score_ptr++;
    }

    return 0;



}

Trocar valores de ponteiros sem usar variavel temporaria

#include <iostream>
#include <string>


using namespace std;


int numero1 = 20;


int numero2 = 30;


void swap_value(int* ptr1, int* ptr2){

    *ptr1 = *ptr1 + *ptr2;
    *ptr2 = *ptr1 - *ptr2;
    *ptr1 = *ptr1 - *ptr2;


    return;
}


int main(){


    int num = 10;
    int num2 = 5;
    int *num_ptr = &num;
    int *num2_ptr = &num2;


    cout << "Ponteiro 1 igual a " << *num_ptr << " Ponteiro 2 " << *num2_ptr << endl;


    swap_value(num2_ptr,num_ptr);
    cout << "Ponteiro 1 igual a " << *num_ptr << " Ponteiro 2 " << *num2_ptr << endl;

    return 0;


}
